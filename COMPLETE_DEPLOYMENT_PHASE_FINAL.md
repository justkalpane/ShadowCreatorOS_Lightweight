```javascript
async persistDossierMutation(dossierId, mutationData) {
  const dossierReader = new DossierResultReader();
  const dossiers = dossierReader.getAllDossiers();
  const idx = dossiers.findIndex(d => d.dossier_id === dossierId);
  
  if (idx >= 0) {
    dossiers[idx] = {
      ...dossiers[idx],
      ...mutationData,
      updated_at: new Date().toISOString(),
      _audit_trail: dossiers[idx]._audit_trail || []
    };
    dossiers[idx]._audit_trail.push({
      event: mutationData.event || 'workflow_executed',
      timestamp: new Date().toISOString(),
      data: mutationData
    });
    dossierReader.writeJson(dossierReader.indexPath, { dossiers, last_updated: new Date().toISOString() });
  }
}
```

```javascript
// In packet_result_reader.js - add after workflow completion
async persistNewPackets(workflowId, executionId, packets) {
  const packetReader = new PacketResultReader();
  const index = packetReader.getAllPackets();
  
  packets.forEach(packet => {
    if (!index.find(p => p.packet_id === packet.packet_id)) {
      index.push({
        packet_id: packet.packet_id,
        dossier_ref: packet.dossier_id,
        artifact_family: packet.artifact_family,
        source_workflow: workflowId,
        execution_id: executionId,
        created_at: new Date().toISOString(),
        size_bytes: packet.size_bytes || 0,
        ready_for_review: packet.ready_for_review || false
      });
    }
  });
  
  packetReader.writeJson(packetReader.indexPath, { packets, last_updated: new Date().toISOString() });
}
```

```javascript
// In review_manager.js - aggregateChildResults()
async aggregateChildResults(dossierId, parentWorkflowId) {
  const routeRuns = this.readRouteRuns();
  const childRuns = routeRuns.filter(r => 
    r.dossier_id === dossierId && 
    r.parent_workflow_id === parentWorkflowId
  );
  
  const allSucceeded = childRuns.length > 0 && 
                       childRuns.every(r => r.status === 'COMPLETED' || r.status === 'success');
  
  const parentRun = routeRuns.find(r => 
    r.dossier_id === dossierId && 
    r.workflow_id === parentWorkflowId
  );
  
  if (parentRun) {
    parentRun.status = allSucceeded ? 'COMPLETED' : 'FAILED';
    parentRun.child_results = {
      total: childRuns.length,
      succeeded: childRuns.filter(r => r.status === 'COMPLETED').length,
      failed: childRuns.filter(r => r.status === 'FAILED').length,
      all_succeeded: allSucceeded
    };
    parentRun.updated_at = new Date().toISOString();
    
    // Write back to se_route_runs.json
    this.writeJson(this.routeRunsPath, routeRuns);
  }
  
  return allSucceeded;
}

// Wire into WF-010 completion handler
const orchestrationSuccess = await this.aggregateChildResults(
  dossierId, 
  'WF-010'
);

return {
  status: orchestrationSuccess ? 'success' : 'error',
  dossier_id: dossierId,
  workflow_complete: true,
  child_results_aggregated: true
};
```
